client = Socrata("data.cityofnewyork.us", None) 
api_limit=500000 
counts_df = pd.DataFrame()

for bike_counter in list(ids['id']):
    print(bike_counter)
    n=0 
    loc = 'id=' + str(bike_counter)

    while True:
    
    # In chunks of 500k, get counts for each filtered bike counter 
        results = client.get("uczf-rk3c", limit=api_limit, offset=api_limit*n, where=loc)
        results_df = pd.DataFrame.from_records(results)
        if len(results_df) == 0:
            break 
        
        counts_df = pd.concat([counts_df, results_df], axis=0)
        n = n + 1

counts_df["counts"] = counts_df["counts"].astype('int')
counts_df["time"] = pd.to_datetime(counts_df["date"]).dt.time
counts_df["date"] = pd.to_datetime(counts_df["date"]).dt.date
